Create PROCEDURE [dbo].[Sps_Load_AdmissionTypeShort_Dropdown]
@CompanyId nvarchar(100)='',
@Id int =0
AS
BEGIN
	
	SELECT CONVERT(NVARCHAR(10),Id) as [Value],[Name] AS [Text] FROM Tbl_Mstr_Admission_Master 
	WHERE IsActive=1 And CompanyId=@CompanyId and Id=@Id

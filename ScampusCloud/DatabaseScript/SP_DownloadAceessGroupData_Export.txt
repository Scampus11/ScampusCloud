CREATE PROCEDURE [dbo].[SP_DownloadAceessGroupData_Export]
@Search nvarchar(max)='',
@CompanyId nvarchar(100)=''

AS
BEGIN
		SELECT C.Id,C.Name,AT.Name as CanteenType,C.Description,C.Code from Tbl_Mstr_AccessGroup_Master c
		INNER JOIN Tbl_Mstr_AccessGroup_Type_Master AT ON C.AccessGroupTypeId = AT.Id
		WHERE (C.Id LIKE '%' + @Search + '%' or C.Name LIKE '%' + @Search + '%')
		and C.CompanyId=@CompanyId
		ORDER BY Id ASC
END

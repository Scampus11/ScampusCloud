ALTER PROCEDURE [dbo].[Sps_Load_College_Dropdown]
@CompanyId nvarchar(100)=''	,
@ScampusId int =0
AS
BEGIN
	
	SELECT CONVERT(NVARCHAR(10),Id) as [Value],[Name] AS [Text] FROM Tbl_Mstr_College_Master 
	WHERE IsActive=1 And CompanyId=@CompanyId and ScampusId=@ScampusId

END
